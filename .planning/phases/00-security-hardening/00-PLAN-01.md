<objective>
Add Redis-backed brute-force lockout to the OTP verify endpoint. After 5 failed attempts within
the OTP TTL window, all further attempts return 429 until the key expires. The lockout is enforced
on the authoritative Medusa backend so it cannot be bypassed via the BFF layer.

Purpose: The current verify-otp route returns 401 indefinitely with no counter. An attacker can
try all 1,000,000 six-digit combinations in the 10-minute OTP window. This fix closes that window.

Output:
- otp-store.ts with three new exports (incrementAttempts, getAttempts, deleteAttempts) and MAX_OTP_ATTEMPTS = 5
- verify-otp/route.ts with pre-comparison lockout check, post-mismatch counter increment, post-success counter delete
- Integration test verifying the lockout behaviour end-to-end
</objective>

export async function incrementAttempts(identifier: string): Promise<number> {
  const redis = getRedisClient();
  const key = attemptKey(identifier);
  const count = await redis.incr(key);
  // Always (re)set TTL — idempotent, ensures counter never outlives the OTP window
  await redis.expire(key, OTP_TTL_SECONDS);
  return count;
}

export async function getAttempts(identifier: string): Promise<number> {
  const redis = getRedisClient();
  const val = await redis.get(attemptKey(identifier));
  return val ? parseInt(val, 10) : 0;
}

export async function deleteAttempts(identifier: string): Promise<void> {
  const redis = getRedisClient();
  await redis.del(attemptKey(identifier));
}

```typescript
import { MedusaRequest, MedusaResponse } from "@medusajs/framework/http";
import { Modules } from "@medusajs/framework/utils";
import {
  getOtp,
  deleteOtp,
  getAttempts,
  incrementAttempts,
  deleteAttempts,
  MAX_OTP_ATTEMPTS,
} from "../../../../lib/otp-store";

export async function POST(req: MedusaRequest, res: MedusaResponse) {
  const { identifier, otp, firstName, lastName } = req.body as {
    identifier?: string;
    otp?: string;
    firstName?: string;
    lastName?: string;
  };

  if (!identifier || !otp) {
    return res.status(400).json({ error: "identifier and otp required" });
  }

  // SEC-01: brute-force lockout — check before comparing OTP
  const attempts = await getAttempts(identifier);
  if (attempts >= MAX_OTP_ATTEMPTS) {
    return res.status(429).json({
      error: "Too many failed attempts. Please request a new OTP.",
      attemptsRemaining: 0,
    });
  }

  const entry = await getOtp(identifier);
  if (!entry || Date.now() > entry.expiresAt) {
    return res.status(401).json({ error: "OTP expired or not found" });
  }

  if (entry.otp !== otp) {
    // Increment counter; re-read to give accurate attemptsRemaining in response
    const newCount = await incrementAttempts(identifier);
    const attemptsRemaining = Math.max(0, MAX_OTP_ATTEMPTS - newCount);
    if (attemptsRemaining === 0) {
      return res.status(429).json({
        error: "Too many failed attempts. Please request a new OTP.",
        attemptsRemaining: 0,
      });
    }
    return res.status(401).json({
      error: "Invalid OTP",
      attemptsRemaining,
    });
  }

  // OTP matched — delete both OTP and attempt counter atomically (best-effort)
  await Promise.all([deleteOtp(identifier), deleteAttempts(identifier)]);

  const customerService = req.scope.resolve(Modules.CUSTOMER);
  const isPhone = /^\+?\d+$/.test(identifier);
  const email = isPhone ? null : identifier;
  const phone = isPhone ? identifier : null;

  const existing = email
    ? await customerService.listCustomers({ email })
    : await customerService.listCustomers({ phone });

  let customer = existing[0];

  if (!customer) {
    const resolvedFirst = firstName ?? entry.firstName ?? "";
    const resolvedLast = lastName ?? entry.lastName ?? "";
    const resolvedEmail = email ?? `${identifier.replace(/\D/g, "")}@poshakh.in`;

    customer = await customerService.createCustomers({
      email: resolvedEmail,
      phone: phone ?? undefined,
      first_name: resolvedFirst,
      last_name: resolvedLast,
    });
  }

  res.json({
    customer: {
      id: customer.id,
      email: customer.email,
      phone: customer.phone,
      firstName: customer.first_name,
      lastName: customer.last_name,
    },
  });
}
```

Key logic sequence (per RESEARCH.md architecture pattern):
1. Check getAttempts FIRST — return 429 immediately if already locked
2. Check OTP exists and not expired — return 401 if missing/expired
3. Compare OTP — on mismatch: incrementAttempts, calculate remaining, return 401 (or 429 if now locked)
4. On match: delete both keys in parallel, proceed to customer logic

          identifier,
          otp: "000000",

        expect(lockedRes.data.attemptsRemaining).toEqual(0);
        expect(lockedRes.data.error).toMatch(/too many/i);

        const res1 = await api.post("/store/auth/verify-otp", { identifier, otp: "000000" });
        expect(res1.status).toEqual(401);
        const remaining1 = res1.data.attemptsRemaining;

        const res2 = await api.post("/store/auth/verify-otp", { identifier, otp: "000000" });
        expect(res2.status).toEqual(401);
        const remaining2 = res2.data.attemptsRemaining;

        expect(remaining2).toEqual(remaining1 - 1);
      });
    });
  },
});
