<objective>
Add session expiry to the HMAC cookie implementation. Every new cookie carries an exp timestamp
seven days in the future. verifySignedCookie rejects cookies where exp is in the past.
Old tokens with no exp field are accepted (backward-compatible per MIG-03 design).

Purpose: Currently every issued cookie is permanently valid. A stolen cookie gives indefinite
access. With exp, the maximum exposure window for a stolen cookie is 7 days.

Output:
- session.ts with exp/iat injection in createSignedCookie and expiry guard in verifySignedCookie
- session.test.ts with unit tests covering all four cases (expired, valid, no-exp legacy, bad sig)

IMPORTANT — Do NOT migrate createHmac to crypto.subtle in this plan.
The Node.js crypto.createHmac implementation in session.ts is correct for Phase 0 (route handlers
run in Node runtime). crypto.subtle migration belongs in Phase 3 (BFF-02/BFF-03) when Edge Runtime
middleware is added. Modifying the signature algorithm here would be scope creep.
</objective>

```typescript
import { createHmac } from "crypto";

const SECRET = process.env.MEDUSA_CUSTOMER_SECRET!;
if (!SECRET) throw new Error("MEDUSA_CUSTOMER_SECRET env var is not set");

const SESSION_TTL_MS = 7 * 24 * 60 * 60 * 1000; // 7 days in milliseconds

export function createSignedCookie(data: object): string {
  const now = Date.now();
  const payload = Buffer.from(
    JSON.stringify({
      ...data,
      iat: now,
      exp: now + SESSION_TTL_MS,
    })
  ).toString("base64url");
  const sig = createHmac("sha256", SECRET).update(payload).digest("hex");
  return `${payload}.${sig}`;
}

export function verifySignedCookie(value: string): Record<string, unknown> | null {
  const dotIdx = value.lastIndexOf(".");
  if (dotIdx === -1) return null;
  const payload = value.slice(0, dotIdx);
  const sig = value.slice(dotIdx + 1);
  const expected = createHmac("sha256", SECRET).update(payload).digest("hex");
  if (sig !== expected) return null;
  try {
    const parsed = JSON.parse(Buffer.from(payload, "base64url").toString("utf8"));
    // SEC-03: reject expired tokens; accept tokens with no exp (backward compat per MIG-03)
    if (typeof parsed.exp === "number" && Date.now() > parsed.exp) {
      return null;
    }
    return parsed;
  } catch {
    return null;
  }
}
```

- Guard is `typeof parsed.exp === "number" && Date.now() > parsed.exp` — missing exp passes (false && anything = false)
- SESSION_TTL_MS constant makes the 7-day value self-documenting and testable
- NO changes to import, SECRET guard, HMAC algorithm, or cookie format

```typescript
import { createSignedCookie, verifySignedCookie } from "./session";

let passed = 0;
let failed = 0;

function assert(label: string, condition: boolean) {

    console.log(`  PASS  ${label}`);
    passed++;
  } else {
    console.error(`  FAIL  ${label}`);
    failed++;

// --- Test 1: createSignedCookie injects exp and iat ---
{
  const before = Date.now();
  const token = createSignedCookie({ id: "c_test_1" });
  const after = Date.now();

  const parsed = verifySignedCookie(token) as Record<string, number>;
  assert("createSignedCookie: token verifies successfully", parsed !== null);
  assert("createSignedCookie: iat is within test window", parsed.iat >= before && parsed.iat <= after);
  assert("createSignedCookie: exp = iat + 7 days", parsed.exp === parsed.iat + 7 * 24 * 60 * 60 * 1000);

// --- Test 2: verifySignedCookie returns null for expired token ---
```
